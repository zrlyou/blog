create database blog default charset utf8;  建立数据库blog

create table user(
	uid int unsigned not null primary key auto_increment,
	username varchar(50) not null default '',
	password varchar(32) not null default '',
	signature varchar(100)  not null default '',
	weibo varchar(100) not null default '',
	qq_zone varchar(100) not null default '',
	logintime int(10) unsigned not null default 0,
	loginip varchar(20) not null default '0.0.0.0'
	) engine myisam default charset utf8;

create table blog(
	bid int unsigned not null primary key auto_increment,
	title varchar(100) not null default '',
	time int(10) unsigned not null default 0,
	content text
	) engine myisam auto_increment=100 default charset utf8;

create table log(
	lid int unsigned not null primary key auto_increment,
	username varchar(50) not null default '',
	loginip varchar(20) not null default '',
	logintime int(10) unsigned not null default 0,
	status tinyint(1) unsigned not null default 0
	) engine myisam default charset utf8;
